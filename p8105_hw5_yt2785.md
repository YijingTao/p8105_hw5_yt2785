Fill in missing values with the mean of non-missing values For character
variables, fill in missing values with `virginica`

``` r
iris_na = function(x) {
  if (is.numeric(x)){
    x[is.na(x)] = round(mean(x,na.rm = T))
  return(x)
  }
  if (is.character(x)) {
    x[is.na(x)] = "virginica"
    return(x)
  }}
iris_df = map_df(iris_with_missing, iris_na)
```
